import "dotenv/config";
import fetch from "node-fetch";
import fs from "fs";
import crypto from "crypto";

// ─────────── ACCOUNT ───────────
const ACCOUNT = {
  balance:              parseFloat(process.env.ACCOUNT_BALANCE || "6000"),
  riskPercent:          parseFloat(process.env.RISK_PERCENT    || "0.5"),
  rewardRatio:          parseFloat(process.env.REWARD_RATIO    || "2"),
  maxDailyLossPercent:  4.5,
  maxTotalLossPercent:  9.5,
  maxTradesPerDay:      3,
  maxLossesPerDay:      2,
};

const SYMBOL           = process.env.SYMBOL        || "BTCUSDT";
const PAPER            = process.env.PAPER_TRADING !== "false";
const BINANCE_API_KEY  = process.env.BINANCE_API_KEY || "";
const BINANCE_SECRET   = process.env.BINANCE_SECRET  || "";
const TELEGRAM_TOKEN   = process.env.TELEGRAM_TOKEN  || "";
const TELEGRAM_CHAT_ID = process.env.TELEGRAM_CHAT_ID|| "";
const BASE_URL         = "https://fapi.binance.com";

// ─────────── STATE FILES ───────────
const STATE_FILE  = "state.json";
const TRADES_FILE = "trades.json";

function loadState() {
  try {
    const s     = JSON.parse(fs.readFileSync(STATE_FILE, "utf8"));
    const today = new Date().toISOString().slice(0, 10);
    if (s.date !== today) {
      return { date: today, trades: 0, losses: 0,
               dailyPnL: 0, totalPnL: s.totalPnL || 0 };
    }
    return s;
  } catch {
    return { date: new Date().toISOString().slice(0, 10),
             trades: 0, losses: 0, dailyPnL: 0, totalPnL: 0 };
  }
}

function saveState(s) {
  fs.writeFileSync(STATE_FILE, JSON.stringify(s, null, 2));
}

function saveTrade(t) {
  let arr = [];
  try { arr = JSON.parse(fs.readFileSync(TRADES_FILE, "utf8")); } catch {}
  arr.push(t);
  fs.writeFileSync(TRADES_FILE, JSON.stringify(arr, null, 2));
}

// ─────────── HELPERS ───────────
function log(msg) {
  console.log(`[${new Date().toISOString()}] ${msg}`);
}

async function sendTelegram(msg) {
  if (!TELEGRAM_TOKEN || !TELEGRAM_CHAT_ID) return;
  try {
    await fetch(`https://api.telegram.org/bot${TELEGRAM_TOKEN}/sendMessage`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ chat_id: TELEGRAM_CHAT_ID, text: msg }),
    });
  } catch {}
}

// ─────────── SESSION FILTER ───────────
function isTradingSession() {
  const h = new Date().getUTCHours();
  const m = new Date().getUTCMinutes();
  const t = h + m / 60;
  const london  = t >= 8  && t < 11;
  const newYork = t >= 13 && t < 17;
  if (!london && !newYork) {
    log(`❌ Outside sessions (UTC ${h}:${String(m).padStart(2,"0")})`);
    return false;
  }
  log(`✅ Session active: ${london ? "London" : "New York"}`);
  return true;
}

// ─────────── NEWS FILTER ───────────
function isNewsWindow() {
  const h   = new Date().getUTCHours();
  const m   = new Date().getUTCMinutes();
  const now = h * 60 + m;
  const events = [
    { h: 13, m: 30 },  // NFP / CPI / US data
    { h: 18, m: 0  },  // FOMC
    { h: 19, m: 0  },  // FOMC alternate
  ];
  for (const e of events) {
    const center = e.h * 60 + e.m;
    if (now >= center - 30 && now <= center + 30) {
      log("⚠️  News window — skipping trade.");
      return true;
    }
  }
  return false;
}

// ─────────── RISK GUARD ───────────
function passesRiskGuard(state) {
  const dLimit = ACCOUNT.balance * ACCOUNT.maxDailyLossPercent / 100;
  const tLimit = ACCOUNT.balance * ACCOUNT.maxTotalLossPercent / 100;

  if (state.trades >= ACCOUNT.maxTradesPerDay) {
    log(`🛑 Max trades/day (${state.trades}/${ACCOUNT.maxTradesPerDay})`);
    return false;
  }
  if (state.losses >= ACCOUNT.maxLossesPerDay) {
    log(`🛑 Max losses/day (${state.losses}/${ACCOUNT.maxLossesPerDay})`);
    return false;
  }
  if (state.dailyPnL <= -dLimit) {
    log(`🛑 Daily DD limit: -$${Math.abs(state.dailyPnL).toFixed(2)} / $${dLimit.toFixed(2)}`);
    return false;
  }
  if (state.totalPnL <= -tLimit) {
    log(`🛑 MAX DD HIT: -$${Math.abs(state.totalPnL).toFixed(2)} — STOP ALL TRADING`);
    return false;
  }
  return true;
}

// ─────────── BINANCE SIGNED REQUEST ───────────
function sign(query) {
  return crypto.createHmac("sha256", BINANCE_SECRET).update(query).digest("hex");
}

async function privateRequest(endpoint, method = "POST", params = {}) {
  const query = new URLSearchParams({ ...params, timestamp: Date.now() }).toString();
  const url   = `${BASE_URL}${endpoint}?${query}&signature=${sign(query)}`;
  const res   = await fetch(url, {
    method,
    headers: { "X-MBX-APIKEY": BINANCE_API_KEY },
  });
  return res.json();
}

// ─────────── MARKET DATA ───────────
async function getKlines(limit = 200) {
  const url = `${BASE_URL}/fapi/v1/klines?symbol=${SYMBOL}&interval=15m&limit=${limit}`;
  const res  = await fetch(url);
  const data = await res.json();
  return data.map(k => ({
    open: +k[1], high: +k[2], low: +k[3], close: +k[4], volume: +k[5],
  }));
}

async function getPrice() {
  const res  = await fetch(`${BASE_URL}/fapi/v1/ticker/price?symbol=${SYMBOL}`);
  const data = await res.json();
  return +data.price;
}

async function hasOpenPosition() {
  if (PAPER) return false;
  const data = await privateRequest("/fapi/v2/positionRisk", "GET");
  const pos  = data.find(p => p.symbol === SYMBOL);
  return pos ? Math.abs(+pos.positionAmt) > 0 : false;
}

// ─────────── INDICATORS ───────────
function calcEMA(candles, period = 200) {
  const k = 2 / (period + 1);
  let ema  = candles[0].close;
  for (let i = 1; i < candles.length; i++) {
    ema = candles[i].close * k + ema * (1 - k);
  }
  return ema;
}

function calcVWAP(candles) {
  let tpv = 0, vol = 0;
  for (const c of candles) {
    const tp = (c.high + c.low + c.close) / 3;
    tpv += tp * c.volume;
    vol += c.volume;
  }
  return vol === 0 ? 0 : tpv / vol;
}

function calcATR(candles, period = 14) {
  const trs = [];
  for (let i = 1; i < candles.length; i++) {
    trs.push(Math.max(
      candles[i].high - candles[i].low,
      Math.abs(candles[i].high - candles[i-1].close),
      Math.abs(candles[i].low  - candles[i-1].close),
    ));
  }
  return trs.slice(-period).reduce((a, b) => a + b, 0) / period;
}

function calcORB(candles, bars = 4) {
  const orb = candles.slice(-bars);
  return {
    high: Math.max(...orb.map(c => c.high)),
    low:  Math.min(...orb.map(c => c.low)),
  };
}

function calcPositionSize(entry, stop) {
  const riskUSD = ACCOUNT.balance * (ACCOUNT.riskPercent / 100);
  const riskPer = Math.abs(entry - stop);
  return riskPer === 0 ? 0 : riskUSD / riskPer;
}

// ─────────── EXECUTE TRADE ───────────
async function executeTrade(signal, quantity, stop, target) {
  if (PAPER) { log("📝 PAPER MODE — no real order sent."); return; }

  const side     = signal === "LONG" ? "BUY"  : "SELL";
  const opposite = signal === "LONG" ? "SELL" : "BUY";

  await privateRequest("/fapi/v1/order", "POST", {
    symbol: SYMBOL, side, type: "MARKET", quantity,
  });
  await privateRequest("/fapi/v1/order", "POST", {
    symbol: SYMBOL, side: opposite, type: "STOP_MARKET",
    stopPrice: stop.toFixed(2), closePosition: true,
  });
  await privateRequest("/fapi/v1/order", "POST", {
    symbol: SYMBOL, side: opposite, type: "TAKE_PROFIT_MARKET",
    stopPrice: target.toFixed(2), closePosition: true,
  });

  log("🚀 LIVE TRADE EXECUTED");
}

// ─────────── MAIN ───────────
async function run() {
  log("════════════════════════════════════════════");
  log(`💼 FundedNext ORB+VWAP Bot | $${ACCOUNT.balance} | Risk: ${ACCOUNT.riskPercent}%`);
  log(`💰 Per trade: $${(ACCOUNT.balance * ACCOUNT.riskPercent / 100).toFixed(2)} | Daily limit: $${(ACCOUNT.balance * ACCOUNT.maxDailyLossPercent / 100).toFixed(2)}`);

  // ── Guards ──
  if (!isTradingSession()) return;
  if (isNewsWindow())      return;

  const state = loadState();
  log(`📊 Today: ${state.trades} trades | ${state.losses} losses | PnL: $${state.dailyPnL.toFixed(2)}`);
  if (!passesRiskGuard(state)) return;

  if (await hasOpenPosition()) { log("⚠️  Position already open."); return; }

  // ── Data ──
  const [candles, price] = await Promise.all([getKlines(200), getPrice()]);

  // ── Indicators ──
  const ema  = calcEMA(candles, 200);
  const vwap = calcVWAP(candles);
  const atr  = calcATR(candles, 14);
  const orb  = calcORB(candles, 4);
  const avgVol = candles.slice(-20).reduce((a, b) => a + b.volume, 0) / 20;
  const curVol = candles[candles.length - 1].volume;

  log(`📈 Price: ${price.toFixed(2)} | EMA200: ${ema.toFixed(2)} | VWAP: ${vwap.toFixed(2)} | ATR: ${atr.toFixed(2)}`);
  log(`📦 ORB H: ${orb.high.toFixed(2)} | ORB L: ${orb.low.toFixed(2)} | Vol: ${curVol.toFixed(0)} / Avg: ${avgVol.toFixed(0)}`);

  // ── Filters ──
  if (atr < price * 0.002) { log("⚠️  Low volatility — skip."); return; }
  if (curVol < avgVol)     { log("⚠️  Weak volume — skip.");    return; }

  // ── Signal ──
  let signal = null;
  if (price > ema && price > vwap && price > orb.high) signal = "LONG";
  if (price < ema && price < vwap && price < orb.low)  signal = "SHORT";

  if (!signal) { log("⏳ No setup — waiting."); return; }

  // ── Levels ──
  const stop   = signal === "LONG" ? orb.low : orb.high;
  const risk   = Math.abs(price - stop);
  const target = signal === "LONG"
    ? price + risk * ACCOUNT.rewardRatio
    : price - risk * ACCOUNT.rewardRatio;

  const qty      = calcPositionSize(price, stop);
  const riskUSD  = ACCOUNT.balance * ACCOUNT.riskPercent / 100;
  const gainUSD  = riskUSD * ACCOUNT.rewardRatio;

  log(`🎯 ══ ${signal} SIGNAL ══`);
  log(`   Entry:   $${price.toFixed(2)}`);
  log(`   Stop:    $${stop.toFixed(2)}`);
  log(`   Target:  $${target.toFixed(2)}`);
  log(`   Size:    ${qty.toFixed(6)} units`);
  log(`   Risk:    $${riskUSD.toFixed(2)} (${ACCOUNT.riskPercent}%)`);
  log(`   Reward:  $${gainUSD.toFixed(2)} (${(ACCOUNT.riskPercent * ACCOUNT.rewardRatio).toFixed(1)}%)`);
  log(`   RR:      1:${ACCOUNT.rewardRatio}`);
  log(`   Mode:    ${PAPER ? "PAPER 📝" : "LIVE 🚀"}`);

  // ── Execute ──
  await executeTrade(signal, qty.toFixed(3), stop, target);

  // ── Update state ──
  state.trades += 1;
  saveState(state);

  // ── Log trade ──
  const trade = {
    time: new Date().toISOString(), symbol: SYMBOL, signal,
    entry: price, stop: +stop.toFixed(2), target: +target.toFixed(2),
    size: +qty.toFixed(6), riskUSD: +riskUSD.toFixed(2),
    gainUSD: +gainUSD.toFixed(2), rr: `1:${ACCOUNT.rewardRatio}`,
    paper: PAPER, balance: ACCOUNT.balance,
  };
  saveTrade(trade);

  // ── Telegram ──
  await sendTelegram(
    `🤖 FundedNext Bot\n` +
    `📍 ${signal} ${SYMBOL}\n` +
    `Entry: $${price.toFixed(2)}\n` +
    `Stop:  $${stop.toFixed(2)}\n` +
    `TP:    $${target.toFixed(2)}\n` +
    `Risk:  $${riskUSD.toFixed(2)} | RR 1:${ACCOUNT.rewardRatio}\n` +
    `Mode:  ${PAPER ? "PAPER" : "LIVE"}`
  );

  log("✅ Done.");
}

run().catch(async err => {
  log(`💥 ERROR: ${err.message}`);
  await sendTelegram(`💥 BOT ERROR\n${err.message}`);
  process.exit(1);
});
